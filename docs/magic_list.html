magic_list

This module contains two symbols:

  • list, a type that extends the built-in equivalent
  • L, a pseudo-literal which can be used to create magic lists similarly to built-in ones.

184    def rotate(self, n: int = 1) -> typing_extensions.Self:
185        """
186        Shift the list `n` times to the right. The items that overflow get prepended.
187
188        If `n` is negative, the shift goes to the left.
189
190        .. warning:: The list must be non-empty.
191
192        >>> L[3, 5, 2].rotate()
193        [2, 3, 5]
194        >>> L[3, 5, 2].rotate(2)
195        [5, 2, 3]
196        >>> L[3, 5, 2].rotate(-1)
197        [5, 2, 3]
198        >>> list().rotate()
199        *- TypeError: empty list cannot be rotated -*
200        """
201
202        if not self:
203            raise TypeError("empty list cannot be rotated")
204
205        if n == 0:
206            return self
207
208        returned_list = self.copy()
209
210        if n > 0:
211            xpend_method = returned_list.prepend
212            popped_index = -1
213        else:
214            xpend_method = returned_list.append
215            popped_index = 0
216
217        for _ in range(abs(n)):
218            xpend_method(returned_list.pop(popped_index))
219
220        return returned_list

378        initial_value: _T,

380        """

382        from left to right and return the intermediate values followed by the

456    def flatten(self, *, _base: list[typing.Any] | None = None) -> list[typing.Any]:
457        """
458        Flatten the contents to a 1-dimension list. If the list contains
459        itself, it cannot be flattened and a `ValueError` is raised.
460
461        .. warning:: The list cannot contain recursive elements.
462
463        >>> L[[3, 5, 2], [8, 4, 1], [7, 6, 9]].flatten()
464        [3, 5, 2, 8, 4, 1, 7, 6, 9]
465        >>> list().flatten()
466        []
467        >>> l = list()
468        >>> l.append(l)
469        >>> l.flatten()
470        *- ValueError: cannot flatten list because it contains recursive elements -*
471        """
472
473        err = ValueError("cannot flatten list because it contains recursive elements")
474
475        result: list[typing.Any] = list()
476
477        for item in self:
478            if item is self or item is _base:
479                raise err
480
481            base = self if _base is None else _base
482
483            if isinstance(item, list):
484                result.extend(item.flatten(_base=base))
485            elif isinstance(item, collections.abc.Iterable):
486                try:
487                    result.extend(
488                        list(
489                            typing.cast(collections.abc.Iterable[typing.Any], item),
490                        ).flatten(_base=base),
491                    )
492                except RecursionError:
493                    # a bit dirty but I can't think of any other solution 😅
494                    raise err from None
495            else:
496                result.append(item)
497
498        return result

594        Fill on the left the list with `filler` and return the result.

596        If `filler` is a function, it takes the current list (at the current

609        """

611        if n < 0:
612            raise ValueError("the number of times to fill cannot be negative")
613
614        returned_list = self.copy()
615
616        for _ in range(n):
617            returned_list.prepend(filler(returned_list) if callable(filler) else filler)
618
619        return returned_list

655        self,
656        filler: _T | collections.abc.Callable[[_T, _T], _T],
657    ) -> typing_extensions.Self:
658        """
659        Fill in-between the items with `filler` and return the result.
660
661        If `filler` is a function, it takes the two items surrounding the gap
662        that is about to be filled and produces a new value to be inserted.
663
664        .. warning:: The list must contain at least two items.
665
666        >>> L[3, 5, 2].interleave(0)
667        [3, 0, 5, 0, 2]
668        >>> L[3, 5, 2].interleave(operator.add)
669        [3, 8, 5, 7, 2]
670        >>> L[0].interleave(1)
671        *- ValueError: list has no gap to be filled -*
672        >>> list().interleave(0)
673        *- ValueError: list has no gap to be filled -*
674        """
675
676        if len(self) <= 1:
677            raise ValueError("list has no gap to be filled")
678
679        returned_list = self.__class__([self.head])
680
681        for i in range(1, len(self)):
682            returned_list.append(
683                filler(self[i - 1], self[i]) if callable(filler) else filler,
684            )
685            returned_list.append(self[i])
686
687        return returned_list
